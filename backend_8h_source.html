150 inline struct w_iov * __attribute__((nonnull))
151 alloc_iov(struct warpcore * const w)
152 {
153  struct w_iov * const v = STAILQ_FIRST(&w->iov);
154  ensure(v != 0, "out of spare iovs");
155  STAILQ_REMOVE_HEAD(&w->iov, next);
156  // warn(debug, "allocating spare iov %u", v->idx);
157  v->buf = IDX2BUF(w, v->idx);
158  v->len = w->mtu;
159  return v;
160 }
